/*Write a program in C to check a number is palindrome/Madam number or not*/
// 125!=521 -> not palindrome, 121=121 -> palindrome (interchaning/swapping positions)

#include<stdio.h>
int main ()
{
	int n, m, r, d=0;
	printf ("\nEnter the number: ");
	scanf ("%d", &n); 
	m = n; //duplicating the value of n in m
	while (n!=0)
	{
		//these operations after running while loop condition is true would inter-change(swap) the positions of number, e.g. 125 will be 521.
		r = n % 10;
		d = d * 10 + r;
		n = n / 10;
	}
	if (m == d)
		printf ("\nEntered number: %d & Position interchanged(swapped) number: %d\nAs %d = %d so Enterd number %d is a PALINDROME Number", m, d, m ,d, m);
	else
		printf ("\nEntered number: %d & Position interchanged(swapped) number: %d\nAs %d != %d so Entered number %d is NOT a PALINDROME Number", m, d, m ,d, m);
	return 0;
}
