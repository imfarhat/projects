/*Write a program in C to print the following series's sum*/
// Series: 1+2^2+3^3+4^4+5^5+.....n^n=sum

#include<stdio.h>
#include<math.h>
int main ()
{
	int n, i;
	unsigned long long int sum = 0; // Use a larger data type than int (%d) and long int (%ld) and unsigned do not store negatives
	//Range: -9,223,372,036,854,775,808 to 9,223,372,036,854,775,807 (signed) or 0 to 18,446,744,073,709,551,615 (unsigned)
	printf ("\nEnter the limit or last digit (only base) of series: ");
	scanf ("%d", &n);
	for (i=1; i<=n; i++)
	{
		sum = sum + (unsigned long long int) pow(i, i);
	}
	printf("\nThe sum of the series = %llu", sum);
	return 0;
}
