/*Write a program in C to print the following pattern:
1
2  3
3  4  5 
4  5  6  7
5  6  7  8  9
*/

#include<stdio.h>
int main ()
{
	int n, i, j, k;
	printf ("\nEnter the total number of lines: ");
	scanf ("%d", &n);
	for (i=1; i<=n; i++) //outer for loop i-> maintains line no.
	{
		k = i; // starts the line with value = current line number, starting of line (k) = line no (i)
		printf ("\n");
		for (j=1; j<=i; j++) //inner for loop j-. nos. in each line; using nested for loop, never use same variable in nested and main for loop
		{
			printf ("%d ",k); //placed 1 bar space after each no.; k starts from 1 and continues the nos. and for loop of i changes lines
			k++;
		}
	}
	return 0;
}
