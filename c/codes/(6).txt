/*Write a program in C to print from 1 to n*/
// Using a while (condition) loop

#include<stdio.h>
int main ()
{
	int n, i=1;
	printf ("\nEnter the limit: "); //taking the highest value of series to print; if no incrementing done while condition being true, the loop wil run infinite times
	scanf ("%d", &n); 
	while (i <= n)
	{
		printf("\n%d", i);
		i++; //incrementing value of i by 1 each time while loop runs	
	}	
	return 0;
}
