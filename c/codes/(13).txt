/*Write a program in C to print the following series and its sum*/
//Print Series & Sum: '1+3+5+7+....n=sum'
//As this is an odd series we need to keep in mind if the limit entered is odd or not;
//To overcome the above issue we are running the loop till '1+3+5+7+....last second digit of series+' only and adding 'n' later to complete the series and the sum

#include<stdio.h>
int main ()
{
	int n, i, sum=0;
	printf ("\nEnter the limit of the series: ");
	scanf ("%d", &n); 
	for (i=1; i<n; i+=2) //for calculating sum till last second digit of series; increment done after changing the vlaue in sum and sum changes before checking the condition i<n
	{
		sum = sum +i; // after running this loop till condition is true sum = '1+3+5+7+.....+last second digit of series'
	}
	sum = sum + n; //now sum = '1+3+5+7+.....last second digit of series+n'
	for (i=1; i<n; i+=2) // for printing '+1+3+5+7+.......last second digit of series+'
	{
		printf ("%d + ", i);
	}
	printf ("%d = %d", n, sum); // for printing 'n = sum', n is last digit of series
	return 0;
}
