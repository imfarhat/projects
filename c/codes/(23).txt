/*Write a program in C to print the following series and its sum*/
// Series: 1+2+3^2+4^3+5^4+.....n^n-1=sum

#include<stdio.h>
#include<math.h>
int main ()
{
	int n, i;
	unsigned long long int sum = 0; // Use a larger data type than int (%d) and long int (%ld) and unsigned do not store negatives
	//Range: -9,223,372,036,854,775,808 to 9,223,372,036,854,775,807 (signed) or 0 to 18,446,744,073,709,551,615 (unsigned)
	printf ("\nEnter the limit or last digit (only base) of series: ");
	scanf ("%d", &n);
	for (i=1; i<=n; i++)
	{
		sum = sum + (unsigned long long int) pow(i, i-1); //Calculates the sum of series
	}
	printf ("\n1 + 2");
	for (i=3; i<=n; i++) //for continuing series after '1 + 2'
	{
		printf (" + %d ^ %d", i, i-1);
	}
	printf(" = %llu", sum);
	return 0;
}
