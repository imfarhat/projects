/*Write a program in C to calculate the factorial of a number*/

#include<stdio.h>
int main ()
{
	int n, i, f=1;
	printf ("\nEnter the number to its calculate factorial: ");
	scanf ("%d", &n); 
	for (i=1; i<=n; i++) //incrementing loop by i++ i.e. i=i+1
	{
		f = f * i;
	}
	printf ("\nFactorial of %d is %d", n, f);
	return 0;
}
