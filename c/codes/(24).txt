/*Write a program in C to check a number is prime or not*/
//Used tenary conditional operator in c for if else statements

#include<stdio.h>
int main ()
{
	int n, i, q=0; //q is used as a flag
	printf ("\nEnter the number: ");
	scanf ("%d", &n);
	for (i=2; i<n; i++)
	if (n%i==0) //Since prime no. is not divisble by any other than 1 and itself, so we checking if given no. is divisible by any no. other than 1 & self
	{
		q = 1;
		break; //once if condition is staisfies the brack will act to come out of the loop
	}
	q == 0 ? printf ("\n%d is prime number", n) : printf ("\n%d not a is prime number", n);
	//syntax for tenary conditional operator in c: condition ? if (withou semicolon) : else (put semicolon at last else part)
	return 0;
}
