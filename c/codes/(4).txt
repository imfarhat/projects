/*Write a program in C to to find out the maximum of three integers*/
// Using a fourth variable max to store max value
//Preffered in exams

#include<stdio.h>
int main ()
{
	int a, b, c, max=0;
	printf ("\nEnter the three numbers one by one: "); //taking three inputs in one go
	scanf ("%d %d %d", &a, &b, &c); 
	if (a>max)
	{   
		max=a; //storing the highest value in max
	}
	if (b>max) //do not use else if as it will not go in each if condition and never update the max hence end up giving incorrect output
	{   
		max=b; //storing the highest value in max
	}
	if (c>max) //do not use else if as it will not go in each if condition and never update the max hence end up giving incorrect output
	{   
		max=c; //storing the highest value in max
	} 	
	printf("\n%d is the largest number", max);
	return 0;
}
